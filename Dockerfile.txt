# 1. Base image with Python
FROM python:3.11

# 2. Speed up install by not asking questions
ARG DEBIAN_FRONTEND=noninteractive

# 3. Install graphics & browser dependencies
RUN apt-get update -q && \
    apt-get install -y -qq --no-install-recommends \
      xvfb libxcomposite1 libxdamage1 libatk1.0-0 libasound2 \
      libdbus-1-3 libnspr4 libgbm1 libatk-bridge2.0-0 libcups2 \
      libxkbcommon0 libatspi2.0-0 libnss3

# 4. Copy your dependency file
COPY requirements.txt .

# 5. Install Python packages *and* the Chromium browser
RUN pip3 install -r requirements.txt && \
    playwright install chromium

# 6. Copy your script into the container
COPY daily_deals_agent.py .

# 7. Make the headless browser work
ENV DISPLAY=:99

# 8. Run a virtual framebuffer and your script
CMD Xvfb :99 -screen 0 1024x768x16 & python3 daily_deals_agent.py
